1.  go语言中的slice的容量达到上限后，使用append扩展容量后，cap会扩大2倍

    func main(){
	    s := []int{5}
	    s = append(s, 7)
	    s = append(s, 9)
	    x := append(s, 11)
	    y := append(s, 12)
	   fmt.Println(s, x, y)
    }
    x为 5 ， 7， 9， 12 
    y为 5 ， 7， 9 ,   12
